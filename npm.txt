# バージョン確認
npm -v

# モジュールのインストール(ローカル)
npm install <モジュール名>

# ローカルでインストールする前にpackage.jsonファイルを作成する必要がある。
npm init
# Enterキーを押していき、最後にyesを入力すればOK。
# 面倒な時は以下でOK
npm init -y

# モジュールのインストール(グローバル)
npm install -g <モジュール名>

# モジュールのアンインストール
npm uninstall <モジュール名>

# インストール済みモジュール一覧(ローカル)
npm ls

# npmのアップデート
npm update -g npm
